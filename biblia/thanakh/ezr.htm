עזרא

א,א וּבִשְׁנַת אַחַת לְכוֹרֶשׁ מֶלֶךְ פָּרַס לִכְלוֹת דְּבַר־יְהוָה מִפִּי יִרְמְיָה  הֵעִיר יְהוָה אֶת־רוּחַ כֹּרֶשׁ מֶלֶךְ־פָּרַס וַיַּעֲבֶר־קוֹל בְּכָל־מַלְכוּתוֹ וְגַם־בְּמִכְתָּב לֵאמֹר׃  א,ב כֹּה אָמַר כֹּרֶשׁ מֶלֶךְ פָּרַס כֹּל מַמְלְכוֹת הָאָרֶץ נָתַן לִי יְהוָה אֱלֹהֵי הַשָּׁמָיִם וְהוּא־פָקַד עָלַי לִבְנוֹת־לוֹ בַיִת בִּירוּשָׁלִַם אֲשֶׁר בִּיהוּדָה׃  א,ג מִי־בָכֶם מִכָּל־עַמּוֹ יְהִי אֱלֹהָיו עִמּוֹ וְיַעַל לִירוּשָׁלִַם אֲשֶׁר בִּיהוּדָה וְיִבֶן אֶת־בֵּית יְהוָה אֱלֹהֵי יִשְׂרָאֵל הוּא הָאֱלֹהִים אֲשֶׁר בִּירוּשָׁלִָם׃  א,ד וְכָל־הַנִּשְׁאָר מִכָּל־הַמְּקֹמוֹת אֲשֶׁר הוּא גָר־שָׁם יְנַשְּׂאוּהוּ אַנְשֵׁי מְקֹמוֹ בְּכֶסֶף וּבְזָהָב וּבִרְכוּשׁ וּבִבְהֵמָה עִם־הַנְּדָבָה לְבֵית הָאֱלֹהִים אֲשֶׁר בִּירוּשָׁלִָם׃  א,ה וַיָּקוּמוּ רָאשֵׁי הָאָבוֹת לִיהוּדָה וּבִנְיָמִן וְהַכֹּהֲנִים וְהַלְוִיִּם לְכֹל הֵעִיר הָאֱלֹהִים אֶת־רוּחוֹ לַעֲלוֹת לִבְנוֹת אֶת־בֵּית יְהוָה אֲשֶׁר בִּירוּשָׁלִָם׃  א,ו וְכָל־סְבִיבֹתֵיהֶם חִזְּקוּ בִידֵיהֶם בִּכְלֵי־כֶסֶף בַּזָּהָב בָּרְכוּשׁ וּבַבְּהֵמָה וּבַמִּגְדָּנוֹת לְבַד עַל־כָּל־הִתְנַדֵּב׃  {ס}

א,ז וְהַמֶּלֶךְ כּוֹרֶשׁ הוֹצִיא אֶת־כְּלֵי בֵית־יְהוָה אֲשֶׁר הוֹצִיא נְבוּכַדְנֶצַּר מִירוּשָׁלִַם וַיִּתְּנֵם בְּבֵית אֱלֹהָיו׃  א,ח וַיּוֹצִיאֵם כּוֹרֶשׁ מֶלֶךְ פָּרַס עַל־יַד מִתְרְדָת הַגִּזְבָּר וַיִּסְפְּרֵם לְשֵׁשְׁבַּצַּר הַנָּשִׂיא לִיהוּדָה׃  א,ט וְאֵלֶּה מִסְפָּרָם  אֲגַרְטְלֵי זָהָב שְׁלֹשִׁים אֲגַרְטְלֵי־כֶסֶף אָלֶף מַחֲלָפִים תִּשְׁעָה וְעֶשְׂרִים׃  {ס}

א,י כְּפוֹרֵי זָהָב שְׁלֹשִׁים   {ס}  כְּפוֹרֵי כֶסֶף מִשְׁנִים אַרְבַּע מֵאוֹת וַעֲשָׂרָה כֵּלִים אֲחֵרִים אָלֶף׃  {ס}

א,יא כָּל־כֵּלִים לַזָּהָב וְלַכֶּסֶף חֲמֵשֶׁת אֲלָפִים וְאַרְבַּע מֵאוֹת הַכֹּל הֶעֱלָה שֵׁשְׁבַּצַּר עִם הֵעָלוֹת הַגּוֹלָה מִבָּבֶל לִירוּשָׁלִָם׃  {פ}

ב,א וְאֵלֶּה בְּנֵי הַמְּדִינָה הָעֹלִים מִשְּׁבִי הַגּוֹלָה אֲשֶׁר הֶגְלָה נבוכדנצור (נְבוּכַדְנֶצַּר) מֶלֶךְ־בָּבֶל לְבָבֶל וַיָּשׁוּבוּ לִירוּשָׁלִַם וִיהוּדָה אִישׁ לְעִירוֹ׃  ב,ב אֲשֶׁר־בָּאוּ עִם־זְרֻבָּבֶל יֵשׁוּעַ נְחֶמְיָה שְׂרָיָה רְעֵלָיָה מָרְדֳּכַי בִּלְשָׁן מִסְפָּר בִּגְוַי רְחוּם בַּעֲנָה  מִסְפַּר אַנְשֵׁי עַם יִשְׂרָאֵל׃  {ס}  ב,ג בְּנֵי פַרְעֹשׁ אַלְפַּיִם מֵאָה שִׁבְעִים וּשְׁנָיִם׃  {ס}  ב,ד בְּנֵי שְׁפַטְיָה שְׁלֹשׁ מֵאוֹת שִׁבְעִים וּשְׁנָיִם׃  {ס}  ב,ה בְּנֵי אָרַח שְׁבַע מֵאוֹת חֲמִשָּׁה וְשִׁבְעִים׃  {ס}  ב,ו בְּנֵי־פַחַת מוֹאָב לִבְנֵי יֵשׁוּעַ יוֹאָב אַלְפַּיִם שְׁמֹנֶה מֵאוֹת וּשְׁנֵים עָשָׂר׃  {ס}  ב,ז בְּנֵי עֵילָם אֶלֶף מָאתַיִם חֲמִשִּׁים וְאַרְבָּעָה׃  {ס}  ב,ח בְּנֵי זַתּוּא תְּשַׁע מֵאוֹת וְאַרְבָּעִים וַחֲמִשָּׁה׃  {ס}  ב,ט בְּנֵי זַכָּי שְׁבַע מֵאוֹת וְשִׁשִּׁים׃  {ס}  ב,י בְּנֵי בָנִי שֵׁשׁ מֵאוֹת אַרְבָּעִים וּשְׁנָיִם׃  {ס}  ב,יא בְּנֵי בֵבָי שֵׁשׁ מֵאוֹת עֶשְׂרִים וּשְׁלֹשָׁה׃  {ס}  ב,יב בְּנֵי עַזְגָּד אֶלֶף מָאתַיִם עֶשְׂרִים וּשְׁנָיִם׃  {ס}  ב,יג בְּנֵי אֲדֹנִיקָם שֵׁשׁ מֵאוֹת שִׁשִּׁים וְשִׁשָּׁה׃  {ס}  ב,יד בְּנֵי בִגְוָי אַלְפַּיִם חֲמִשִּׁים וְשִׁשָּׁה׃  {ס}  ב,טו בְּנֵי עָדִין אַרְבַּע מֵאוֹת חֲמִשִּׁים וְאַרְבָּעָה׃  {ס}  ב,טז בְּנֵי־אָטֵר לִיחִזְקִיָּה תִּשְׁעִים וּשְׁמֹנָה׃  {ס}  ב,יז בְּנֵי בֵצָי שְׁלֹשׁ מֵאוֹת עֶשְׂרִים וּשְׁלֹשָׁה׃  {ס}  ב,יח בְּנֵי יוֹרָה מֵאָה וּשְׁנֵים עָשָׂר׃  {ס}  ב,יט בְּנֵי חָשֻׁם מָאתַיִם עֶשְׂרִים וּשְׁלֹשָׁה׃  {ס}  ב,כ בְּנֵי גִבָּר תִּשְׁעִים וַחֲמִשָּׁה׃  {ס}  ב,כא בְּנֵי בֵית־לָחֶם מֵאָה עֶשְׂרִים וּשְׁלֹשָׁה׃  {ס}  ב,כב אַנְשֵׁי נְטֹפָה חֲמִשִּׁים וְשִׁשָּׁה׃  ב,כג אַנְשֵׁי עֲנָתוֹת מֵאָה עֶשְׂרִים וּשְׁמֹנָה׃  {ס}  ב,כד בְּנֵי עַזְמָוֶת אַרְבָּעִים וּשְׁנָיִם׃  {ס}  ב,כה בְּנֵי קִרְיַת עָרִים כְּפִירָה וּבְאֵרוֹת שְׁבַע מֵאוֹת וְאַרְבָּעִים וּשְׁלֹשָׁה׃  {ס}  ב,כו בְּנֵי הָרָמָה וָגָבַע שֵׁשׁ מֵאוֹת עֶשְׂרִים וְאֶחָד׃  {ס}  ב,כז אַנְשֵׁי מִכְמָס מֵאָה עֶשְׂרִים וּשְׁנָיִם׃  {ס}  ב,כח אַנְשֵׁי בֵית־אֵל וְהָעָי מָאתַיִם עֶשְׂרִים וּשְׁלֹשָׁה׃  {ס}  ב,כט בְּנֵי נְבוֹ חֲמִשִּׁים וּשְׁנָיִם׃  {ס}  ב,ל בְּנֵי מַגְבִּישׁ מֵאָה חֲמִשִּׁים וְשִׁשָּׁה׃  {ס}  ב,לא בְּנֵי עֵילָם אַחֵר אֶלֶף מָאתַיִם חֲמִשִּׁים וְאַרְבָּעָה׃  {ס}  ב,לב בְּנֵי חָרִם שְׁלֹשׁ מֵאוֹת וְעֶשְׂרִים׃  {ס}  ב,לג בְּנֵי־לֹד חָדִיד וְאוֹנוֹ שְׁבַע מֵאוֹת עֶשְׂרִים וַחֲמִשָּׁה׃  {ס}  ב,לד בְּנֵי יְרֵחוֹ שְׁלֹשׁ מֵאוֹת אַרְבָּעִים וַחֲמִשָּׁה׃  {ס}  ב,לה בְּנֵי סְנָאָה שְׁלֹשֶׁת אֲלָפִים וְשֵׁשׁ מֵאוֹת וּשְׁלֹשִׁים׃  {ס}  ב,לו הַכֹּהֲנִים  בְּנֵי יְדַעְיָה לְבֵית יֵשׁוּעַ תְּשַׁע מֵאוֹת שִׁבְעִים וּשְׁלֹשָׁה׃  {ס}  ב,לז בְּנֵי אִמֵּר אֶלֶף חֲמִשִּׁים וּשְׁנָיִם׃  {ס}  ב,לח בְּנֵי פַשְׁחוּר אֶלֶף מָאתַיִם אַרְבָּעִים וְשִׁבְעָה׃  {ס}  ב,לט בְּנֵי חָרִם אֶלֶף וְשִׁבְעָה עָשָׂר׃  {ס}  ב,מ הַלְוִיִּם  בְּנֵי־יֵשׁוּעַ וְקַדְמִיאֵל לִבְנֵי הוֹדַוְיָה שִׁבְעִים וְאַרְבָּעָה׃  {ס}  ב,מא הַמְשֹׁרְרִים בְּנֵי אָסָף מֵאָה עֶשְׂרִים וּשְׁמֹנָה׃  {פ}

ב,מב בְּנֵי הַשֹּׁעֲרִים בְּנֵי־שַׁלּוּם בְּנֵי־אָטֵר בְּנֵי־טַלְמֹן בְּנֵי־עַקּוּב בְּנֵי חֲטִיטָא בְּנֵי שֹׁבָי הַכֹּל מֵאָה שְׁלֹשִׁים וְתִשְׁעָה׃  {פ}

ב,מג הַנְּתִינִים  {ס}  בְּנֵי־צִיחָא  {ס}  בְנֵי־חֲשׂוּפָא  {ס}  בְּנֵי  {ר}
טַבָּעוֹת׃  {ס}  ב,מד בְּנֵי־קֵרֹס  {ס}  בְּנֵי־סִיעֲהָא  {ס}  בְּנֵי  {ר}
פָדוֹן׃  {ס}  ב,מה בְּנֵי־לְבָנָה  {ס}  בְנֵי־חֲגָבָה  {ס}  בְּנֵי  {ר}
עַקּוּב׃  {ס}  ב,מו בְּנֵי־חָגָב  {ס}  בְּנֵי־שמלי (שַׁלְמַי)  {ס}  בְּנֵי  {ר}
חָנָן׃  {ס}  ב,מז בְּנֵי־גִדֵּל  {ס}  בְּנֵי־גַחַר  {ס}  בְּנֵי  {ר}
רְאָיָה׃  {ס}  ב,מח בְּנֵי־רְצִין  {ס}  בְּנֵי־נְקוֹדָא  {ס}  בְּנֵי  {ר}
גַזָּם׃  {ס}  ב,מט בְּנֵי־עֻזָּא  {ס}  בְנֵי־פָסֵחַ  {ס}  בְּנֵי  {ר}
בֵסָי׃  {ס}  ב,נ בְּנֵי־אַסְנָה  {ס}  בְנֵי־מְעוּנִים  {ס}  בְּנֵי  {ר}
נפיסים (נְפוּסִים)׃  {ס}  ב,נא בְּנֵי־בַקְבּוּק  {ס}  בְּנֵי־חֲקוּפָא  {ס}  בְּנֵי  {ר}
חַרְחוּר׃  {ס}  ב,נב בְּנֵי־בַצְלוּת  {ס}  בְּנֵי־מְחִידָא  {ס}  בְּנֵי  {ר}
חַרְשָׁא׃  {ס}  ב,נג בְּנֵי־בַרְקוֹס  {ס}  בְּנֵי־סִיסְרָא  {ס}  בְּנֵי  {ר}
תָמַח׃  {ס}  ב,נד בְּנֵי נְצִיחַ  {ס}  בְּנֵי חֲטִיפָא׃  {ס}  ב,נה בְּנֵי  {ר}
עַבְדֵי שְׁלֹמֹה  {ס}  בְּנֵי־סֹטַי  {ס}  בְּנֵי־הַסֹּפֶרֶת  {ס}  בְּנֵי  {ר}
פְרוּדָא׃  {ס}  ב,נו בְּנֵי־יַעְלָה  {ס}  בְנֵי־דַרְקוֹן  {ס}  בְּנֵי  {ר}
גִדֵּל׃  {ס}  ב,נז בְּנֵי שְׁפַטְיָה  {ס}  בְנֵי־חַטִּיל  {ס}  בְּנֵי  {ר}
פֹּכֶרֶת הַצְּבָיִים   {ס}  בְּנֵי אָמִי׃  ב,נח כָּל־הַנְּתִינִים וּבְנֵי עַבְדֵי שְׁלֹמֹה  שְׁלֹשׁ מֵאוֹת תִּשְׁעִים וּשְׁנָיִם׃  {ס}

ב,נט וְאֵלֶּה הָעֹלִים מִתֵּל מֶלַח תֵּל חַרְשָׁא כְּרוּב אַדָּן אִמֵּר וְלֹא יָכְלוּ לְהַגִּיד בֵּית־אֲבוֹתָם וְזַרְעָם אִם מִיִּשְׂרָאֵל הֵם׃  ב,ס בְּנֵי־דְלָיָה בְנֵי־טוֹבִיָּה בְּנֵי נְקוֹדָא שֵׁשׁ מֵאוֹת חֲמִשִּׁים וּשְׁנָיִם׃  {ס}

ב,סא וּמִבְּנֵי הַכֹּהֲנִים בְּנֵי חֳבַיָּה בְּנֵי הַקּוֹץ  בְּנֵי בַרְזִלַּי אֲשֶׁר לָקַח מִבְּנוֹת בַּרְזִלַּי הַגִּלְעָדִי אִשָּׁה וַיִּקָּרֵא עַל־שְׁמָם׃  ב,סב אֵלֶּה בִּקְשׁוּ כְתָבָם הַמִּתְיַחְשִׂים וְלֹא נִמְצָאוּ וַיְגֹאֲלוּ מִן־הַכְּהֻנָּה׃  ב,סג וַיֹּאמֶר הַתִּרְשָׁתָא לָהֶם אֲשֶׁר לֹא־יֹאכְלוּ מִקֹּדֶשׁ הַקֳּדָשִׁים עַד עֲמֹד כֹּהֵן לְאוּרִים וּלְתֻמִּים׃  ב,סד כָּל־הַקָּהָל כְּאֶחָד אַרְבַּע רִבּוֹא אַלְפַּיִם שְׁלֹשׁ־מֵאוֹת שִׁשִּׁים׃  ב,סה מִלְּבַד עַבְדֵיהֶם וְאַמְהֹתֵיהֶם אֵלֶּה שִׁבְעַת אֲלָפִים שְׁלֹשׁ מֵאוֹת שְׁלֹשִׁים וְשִׁבְעָה וְלָהֶם מְשֹׁרְרִים וּמְשֹׁרְרוֹת מָאתָיִם׃  ב,סו סוּסֵיהֶם שְׁבַע מֵאוֹת שְׁלֹשִׁים וְשִׁשָּׁה פִּרְדֵיהֶם מָאתַיִם אַרְבָּעִים וַחֲמִשָּׁה׃  ב,סז גְּמַלֵּיהֶם אַרְבַּע מֵאוֹת שְׁלֹשִׁים וַחֲמִשָּׁה חֲמֹרִים שֵׁשֶׁת אֲלָפִים שְׁבַע מֵאוֹת וְעֶשְׂרִים׃  {פ}

ב,סח וּמֵרָאשֵׁי הָאָבוֹת בְּבוֹאָם לְבֵית יְהוָה אֲשֶׁר בִּירוּשָׁלִָם הִתְנַדְּבוּ לְבֵית הָאֱלֹהִים לְהַעֲמִידוֹ עַל־מְכוֹנוֹ׃  ב,סט כְּכֹחָם נָתְנוּ לְאוֹצַר הַמְּלָאכָה זָהָב דַּרְכְּמוֹנִים שֵׁשׁ־רִבֹּאות וָאֶלֶף  {ס}  וְכֶסֶף מָנִים חֲמֵשֶׁת אֲלָפִים וְכָתְנֹת כֹּהֲנִים מֵאָה׃  {ס}  ב,ע וַיֵּשְׁבוּ הַכֹּהֲנִים וְהַלְוִיִּם וּמִן־הָעָם וְהַמְשֹׁרְרִים וְהַשּׁוֹעֲרִים וְהַנְּתִינִים בְּעָרֵיהֶם וְכָל־יִשְׂרָאֵל בְּעָרֵיהֶם׃  {ס}  ג,א וַיִּגַּע הַחֹדֶשׁ הַשְּׁבִיעִי וּבְנֵי יִשְׂרָאֵל בֶּעָרִים  {ס}  וַיֵּאָסְפוּ הָעָם כְּאִישׁ אֶחָד אֶל־יְרוּשָׁלִָם׃  {ס}

ג,ב וַיָּקָם יֵשׁוּעַ בֶּן־יוֹצָדָק וְאֶחָיו הַכֹּהֲנִים וּזְרֻבָּבֶל בֶּן־שְׁאַלְתִּיאֵל וְאֶחָיו וַיִּבְנוּ אֶת־מִזְבַּח אֱלֹהֵי יִשְׂרָאֵל לְהַעֲלוֹת עָלָיו עֹלוֹת כַּכָּתוּב בְּתוֹרַת מֹשֶׁה אִישׁ־הָאֱלֹהִים׃  ג,ג וַיָּכִינוּ הַמִּזְבֵּחַ עַל־מְכוֹנֹתָיו כִּי בְּאֵימָה עֲלֵיהֶם מֵעַמֵּי הָאֲרָצוֹת ויעל (וַיַּעֲלוּ) עָלָיו עֹלוֹת לַיהוָה עֹלוֹת לַבֹּקֶר וְלָעָרֶב׃  ג,ד וַיַּעֲשׂוּ אֶת־חַג הַסֻּכּוֹת כַּכָּתוּב וְעֹלַת יוֹם בְּיוֹם בְּמִסְפָּר כְּמִשְׁפַּט דְּבַר־יוֹם בְּיוֹמוֹ׃  ג,ה וְאַחֲרֵי־כֵן עֹלַת תָּמִיד וְלֶחֳדָשִׁים וּלְכָל־מוֹעֲדֵי יְהוָה הַמְקֻדָּשִׁים וּלְכֹל מִתְנַדֵּב נְדָבָה לַיהוָה׃  ג,ו מִיּוֹם אֶחָד לַחֹדֶשׁ הַשְּׁבִיעִי הֵחֵלּוּ לְהַעֲלוֹת עֹלוֹת לַיהוָה וְהֵיכַל יְהוָה לֹא יֻסָּד׃  ג,ז וַיִּתְּנוּ־כֶסֶף לַחֹצְבִים וְלֶחָרָשִׁים וּמַאֲכָל וּמִשְׁתֶּה וָשֶׁמֶן לַצִּדֹנִים וְלַצֹּרִים לְהָבִיא עֲצֵי אֲרָזִים מִן־הַלְּבָנוֹן אֶל־יָם יָפוֹא כְּרִשְׁיוֹן כּוֹרֶשׁ מֶלֶךְ־פָּרַס עֲלֵיהֶם׃  {פ}

ג,ח וּבַשָּׁנָה הַשֵּׁנִית לְבוֹאָם אֶל־בֵּית הָאֱלֹהִים לִירוּשָׁלִַם בַּחֹדֶשׁ הַשֵּׁנִי הֵחֵלּוּ זְרֻבָּבֶל בֶּן־שְׁאַלְתִּיאֵל וְיֵשׁוּעַ בֶּן־יוֹצָדָק וּשְׁאָר אֲחֵיהֶם הַכֹּהֲנִים וְהַלְוִיִּם וְכָל־הַבָּאִים מֵהַשְּׁבִי יְרוּשָׁלִַם וַיַּעֲמִידוּ אֶת־הַלְוִיִּם מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה לְנַצֵּחַ עַל־מְלֶאכֶת בֵּית־יְהוָה׃  {פ}

ג,ט וַיַּעֲמֹד יֵשׁוּעַ בָּנָיו וְאֶחָיו קַדְמִיאֵל וּבָנָיו בְּנֵי־יְהוּדָה כְּאֶחָד לְנַצֵּחַ עַל־עֹשֵׂה הַמְּלָאכָה בְּבֵית הָאֱלֹהִים  {ס}  בְּנֵי חֵנָדָד בְּנֵיהֶם וַאֲחֵיהֶם הַלְוִיִּם׃  ג,י וְיִסְּדוּ הַבֹּנִים אֶת־הֵיכַל יְהוָה וַיַּעֲמִידוּ הַכֹּהֲנִים מְלֻבָּשִׁים בַּחֲצֹצְרוֹת וְהַלְוִיִּם בְּנֵי־אָסָף בַּמְצִלְתַּיִם לְהַלֵּל אֶת־יְהוָה עַל־יְדֵי דָּוִיד מֶלֶךְ־יִשְׂרָאֵל׃  ג,יא וַיַּעֲנוּ בְּהַלֵּל וּבְהוֹדֹת לַיהוָה כִּי טוֹב כִּי־לְעוֹלָם חַסְדּוֹ עַל־יִשְׂרָאֵל וְכָל־הָעָם הֵרִיעוּ תְרוּעָה גְדוֹלָה בְהַלֵּל לַיהוָה עַל הוּסַד בֵּית־יְהוָה׃  {ס}

ג,יב וְרַבִּים מֵהַכֹּהֲנִים וְהַלְוִיִּם וְרָאשֵׁי הָאָבוֹת הַזְּקֵנִים אֲשֶׁר רָאוּ אֶת־הַבַּיִת הָרִאשׁוֹן בְּיָסְדוֹ זֶה הַבַּיִת בְּעֵינֵיהֶם בֹּכִים בְּקוֹל גָּדוֹל וְרַבִּים בִּתְרוּעָה בְשִׂמְחָה לְהָרִים קוֹל׃  ג,יג וְאֵין הָעָם מַכִּירִים קוֹל תְּרוּעַת הַשִּׂמְחָה לְקוֹל בְּכִי הָעָם  כִּי הָעָם מְרִיעִים תְּרוּעָה גְדוֹלָה וְהַקּוֹל נִשְׁמַע עַד־לְמֵרָחוֹק׃  {פ}

ד,א וַיִּשְׁמְעוּ צָרֵי יְהוּדָה וּבִנְיָמִן  כִּי־בְנֵי הַגּוֹלָה בּוֹנִים הֵיכָל לַיהוָה אֱלֹהֵי יִשְׂרָאֵל׃  ד,ב וַיִּגְּשׁוּ אֶל־זְרֻבָּבֶל וְאֶל־רָאשֵׁי הָאָבוֹת וַיֹּאמְרוּ לָהֶם נִבְנֶה עִמָּכֶם כִּי כָכֶם נִדְרוֹשׁ לֵאלֹהֵיכֶם ולא (וְלוֹ) אֲנַחְנוּ זֹבְחִים מִימֵי אֵסַר חַדֹּן מֶלֶךְ אַשּׁוּר הַמַּעֲלֶה אֹתָנוּ פֹּה׃  ד,ג וַיֹּאמֶר לָהֶם זְרֻבָּבֶל וְיֵשׁוּעַ וּשְׁאָר רָאשֵׁי הָאָבוֹת לְיִשְׂרָאֵל לֹא־לָכֶם וָלָנוּ לִבְנוֹת בַּיִת לֵאלֹהֵינוּ  כִּי אֲנַחְנוּ יַחַד נִבְנֶה לַיהוָה אֱלֹהֵי יִשְׂרָאֵל כַּאֲשֶׁר צִוָּנוּ הַמֶּלֶךְ כּוֹרֶשׁ מֶלֶךְ־פָּרָס׃  ד,ד וַיְהִי עַם־הָאָרֶץ מְרַפִּים יְדֵי עַם־יְהוּדָה ומבלהים (וּמְבַהֲלִים) אוֹתָם לִבְנוֹת׃  ד,ה וְסֹכְרִים עֲלֵיהֶם יוֹעֲצִים לְהָפֵר עֲצָתָם כָּל־יְמֵי כּוֹרֶשׁ מֶלֶךְ פָּרַס וְעַד־מַלְכוּת דָּרְיָוֶשׁ מֶלֶךְ־פָּרָס׃  ד,ו וּבְמַלְכוּת אֲחַשְׁוֵרוֹשׁ בִּתְחִלַּת מַלְכוּתוֹ כָּתְבוּ שִׂטְנָה עַל־יֹשְׁבֵי יְהוּדָה וִירוּשָׁלִָם׃  {ס}

ד,ז וּבִימֵי אַרְתַּחְשַׁשְׂתָּא כָּתַב בִּשְׁלָם מִתְרְדָת טָבְאֵל וּשְׁאָר כְּנָו‍ֹתָו עַל־אַרְתַּחְשַׁשְׂתְּא מֶלֶךְ פָּרָס וּכְתָב הַנִּשְׁתְּוָן כָּתוּב אֲרָמִית וּמְתֻרְגָּם אֲרָמִית׃  {פ}

ד,ח רְחוּם בְּעֵל־טְעֵם וְשִׁמְשַׁי סָפְרָא כְּתַבוּ אִגְּרָה חֲדָה עַל־יְרוּשְׁלֶם לְאַרְתַּחְשַׁשְׂתְּא מַלְכָּא כְּנֵמָא׃  ד,ט אֱדַיִן רְחוּם בְּעֵל־טְעֵם וְשִׁמְשַׁי סָפְרָא וּשְׁאָר כְּנָוָתְהוֹן דִּינָיֵא וַאֲפַרְסַתְכָיֵא טַרְפְּלָיֵא אֲפָרְסָיֵא ארכוי (אַרְכְּוָיֵא) בָבְלָיֵא שׁוּשַׁנְכָיֵא דהוא (דֶּהָיֵא) עֵלְמָיֵא׃  ד,י וּשְׁאָר אֻמַּיָּא דִּי הַגְלִי אָסְנַפַּר רַבָּא וְיַקִּירָא וְהוֹתֵב הִמּוֹ בְּקִרְיָה דִּי שָׁמְרָיִן וּשְׁאָר עֲבַר־נַהֲרָה וּכְעֶנֶת׃  ד,יא דְּנָה פַּרְשֶׁגֶן אִגַּרְתָּא דִּי שְׁלַחוּ עֲלוֹהִי עַל־אַרְתַּחְשַׁשְׂתְּא מַלְכָּא עבדיך (עַבְדָךְ) אֱנָשׁ עֲבַר־נַהֲרָה וּכְעֶנֶת׃  {פ}

ד,יב יְדִיעַ לֶהֱוֵא לְמַלְכָּא דִּי יְהוּדָיֵא דִּי סְלִקוּ מִן־לְוָתָךְ עֲלֶינָא אֲתוֹ לִירוּשְׁלֶם קִרְיְתָא מָרָדְתָּא וּבִאישְׁתָּא בָּנַיִן ושורי אשכללו (וְשׁוּרַיָּא שַׁכְלִלוּ) וְאֻשַּׁיָּא יַחִיטוּ׃  ד,יג כְּעַן יְדִיעַ לֶהֱוֵא לְמַלְכָּא דִּי הֵן קִרְיְתָא דָךְ תִּתְבְּנֵא וְשׁוּרַיָּא יִשְׁתַּכְלְלוּן מִנְדָּה־בְלוֹ וַהֲלָךְ לָא יִנְתְּנוּן וְאַפְּתֹם מַלְכִים תְּהַנְזִק׃  ד,יד כְּעַן כָּל־קֳבֵל דִּי־מְלַח הֵיכְלָא מְלַחְנָא וְעַרְוַת מַלְכָּא לָא אֲרִיךְ־לַנָא לְמֶחֱזֵא עַל־דְּנָה שְׁלַחְנָא וְהוֹדַעְנָא לְמַלְכָּא׃  ד,טו דִּי יְבַקַּר בִּסְפַר־דָּכְרָנַיָּא דִּי אֲבָהָתָךְ וּתְהַשְׁכַּח בִּסְפַר דָּכְרָנַיָּא וְתִנְדַּע דִּי קִרְיְתָא דָךְ קִרְיָא מָרָדָא וּמְהַנְזְקַת מַלְכִין וּמְדִנָן וְאֶשְׁתַּדּוּר עָבְדִין בְּגַוַּהּ מִן־יוֹמָת עָלְמָא עַל־דְּנָה קִרְיְתָא דָךְ הָחָרְבַת׃  ד,טז מְהוֹדְעִין אֲנַחְנָה לְמַלְכָּא דִּי הֵן קִרְיְתָא דָךְ תִּתְבְּנֵא וְשׁוּרַיָּה יִשְׁתַּכְלְלוּן  לָקֳבֵל דְּנָה חֲלָק בַּעֲבַר נַהֲרָא לָא אִיתַי לָךְ׃  {פ}

ד,יז פִּתְגָמָא שְׁלַח מַלְכָּא עַל־רְחוּם בְּעֵל־טְעֵם וְשִׁמְשַׁי סָפְרָא וּשְׁאָר כְּנָוָתְהוֹן דִּי יָתְבִין בְּשָׁמְרָיִן וּשְׁאָר עֲבַר־נַהֲרָה שְׁלָם וּכְעֶת׃  {ס}

ד,יח נִשְׁתְּוָנָא דִּי שְׁלַחְתּוּן עֲלֶינָא מְפָרַשׁ קֱרִי קָדָמָי׃  ד,יט וּמִנִּי שִׂים טְעֵם וּבַקַּרוּ וְהַשְׁכַּחוּ דִּי קִרְיְתָא דָךְ מִן־יוֹמָת עָלְמָא עַל־מַלְכִין מִתְנַשְּׂאָה וּמְרַד וְאֶשְׁתַּדּוּר מִתְעֲבֶד־בַּהּ׃  ד,כ וּמַלְכִין תַּקִּיפִין הֲווֹ עַל־יְרוּשְׁלֶם וְשַׁלִּיטִין בְּכֹל עֲבַר נַהֲרָה וּמִדָּה בְלוֹ וַהֲלָךְ מִתְיְהֵב לְהוֹן׃  ד,כא כְּעַן שִׂימוּ טְּעֵם לְבַטָּלָא גֻּבְרַיָּא אִלֵּךְ וְקִרְיְתָא דָךְ לָא תִתְבְּנֵא עַד־מִנִּי טַעְמָא יִתְּשָׂם׃  ד,כב וּזְהִירִין הֱווֹ שָׁלוּ לְמֶעְבַּד עַל־דְּנָה לְמָה יִשְׂגֵּא חֲבָלָא לְהַנְזָקַת מַלְכִין׃  {ס}

ד,כג אֱדַיִן מִן־דִּי פַּרְשֶׁגֶן נִשְׁתְּוָנָא דִּי אַרְתַּחְשַׁשְׂתְּא מַלְכָּא קֱרִי קֳדָם־רְחוּם וְשִׁמְשַׁי סָפְרָא וּכְנָוָתְהוֹן אֲזַלוּ בִבְהִילוּ לִירוּשְׁלֶם עַל־יְהוּדָיֵא וּבַטִּלוּ הִמּוֹ בְּאֶדְרָע וְחָיִל׃  {ס}

ד,כד בֵּאדַיִן בְּטֵלַת עֲבִידַת בֵּית־אֱלָהָא דִּי בִּירוּשְׁלֶם וַהֲוָת בָּטְלָא עַד שְׁנַת תַּרְתֵּין לְמַלְכוּת דָּרְיָוֶשׁ מֶלֶךְ־פָּרָס׃  {פ}

ה,א וְהִתְנַבִּי חַגַּי נְבִיָּאה וּזְכַרְיָה בַר־עִדּוֹא נְבִיַּאיָּא עַל־יְהוּדָיֵא דִּי בִיהוּד וּבִירוּשְׁלֶם בְּשֻׁם אֱלָהּ יִשְׂרָאֵל עֲלֵיהוֹן׃  {ס}

ה,ב בֵּאדַיִן קָמוּ זְרֻבָּבֶל בַּר־שְׁאַלְתִּיאֵל וְיֵשׁוּעַ בַּר־יוֹצָדָק וְשָׁרִיו לְמִבְנֵא בֵּית אֱלָהָא דִּי בִירוּשְׁלֶם וְעִמְּהוֹן נְבִיַּאיָּא דִי־אֱלָהָא מְסָעֲדִין לְהוֹן׃  {פ}

ה,ג בֵּהּ־זִמְנָא אֲתָה עֲלֵיהוֹן תַּתְּנַי פַּחַת עֲבַר־נַהֲרָה וּשְׁתַר בּוֹזְנַי וּכְנָוָתְהוֹן וְכֵן אָמְרִין לְהֹם מַן־שָׂם לְכֹם טְעֵם בַּיְתָא דְנָה לִבְּנֵא וְאֻשַּׁרְנָא דְנָה לְשַׁכְלָלָה׃  {ס}

ה,ד אֱדַיִן כְּנֵמָא אֲמַרְנָא לְּהֹם  מַן־אִנּוּן שְׁמָהָת גֻּבְרַיָּא דִּי־דְנָה בִנְיָנָא בָּנַיִן׃  ה,ה וְעֵין אֱלָהֲהֹם הֲוָת עַל־שָׂבֵי יְהוּדָיֵא וְלָא־בַטִּלוּ הִמּוֹ עַד־טַעְמָא לְדָרְיָוֶשׁ יְהָךְ וֶאֱדַיִן יְתִיבוּן נִשְׁתְּוָנָא עַל־דְּנָה׃  {פ}

ה,ו פַּרְשֶׁגֶן אִגַּרְתָּא דִּי־שְׁלַח תַּתְּנַי פַּחַת עֲבַר־נַהֲרָה וּשְׁתַר בּוֹזְנַי וּכְנָוָתֵהּ אֲפַרְסְכָיֵא דִּי בַּעֲבַר נַהֲרָה עַל־דָּרְיָוֶשׁ מַלְכָּא׃  ה,ז פִּתְגָמָא שְׁלַחוּ עֲלוֹהִי וְכִדְנָה כְּתִיב בְּגַוֵּהּ לְדָרְיָוֶשׁ מַלְכָּא שְׁלָמָא כֹלָּא׃  {ס}

ה,ח יְדִיעַ לֶהֱוֵא לְמַלְכָּא דִּי־אֲזַלְנָא לִיהוּד מְדִינְתָּא לְבֵית אֱלָהָא רַבָּא וְהוּא מִתְבְּנֵא אֶבֶן גְּלָל וְאָע מִתְּשָׂם בְּכֻתְלַיָּא וַעֲבִידְתָּא דָךְ אָסְפַּרְנָא מִתְעַבְדָא וּמַצְלַח בְּיֶדְהֹם׃  {ס}

ה,ט אֱדַיִן שְׁאֵלְנָא לְשָׂבַיָּא אִלֵּךְ כְּנֵמָא אֲמַרְנָא לְּהֹם  מַן־שָׂם לְכֹם טְעֵם בַּיְתָא דְנָה לְמִבְנְיָה וְאֻשַּׁרְנָא דְנָה לְשַׁכְלָלָה׃  ה,י וְאַף שְׁמָהָתְהֹם שְׁאֵלְנָא לְהֹם לְהוֹדָעוּתָךְ דִּי נִכְתֻּב שֻׁם־גֻּבְרַיָּא דִּי בְרָאשֵׁיהֹם׃  {ס}

ה,יא וּכְנֵמָא פִתְגָמָא הֲתִיבוּנָא לְמֵמַר  אֲנַחְנָא הִמּוֹ עַבְדוֹהִי דִי־אֱלָהּ שְׁמַיָּא וְאַרְעָא וּבָנַיִן בַּיְתָא דִּי־הֲוָא בְנֵה מִקַּדְמַת דְּנָה שְׁנִין שַׂגִּיאָן וּמֶלֶךְ לְיִשְׂרָאֵל רַב בְּנָהִי וְשַׁכְלְלֵהּ׃  ה,יב לָהֵן מִן־דִּי הַרְגִּזוּ אֲבָהָתַנָא לֶאֱלָהּ שְׁמַיָּא יְהַב הִמּוֹ בְּיַד נְבוּכַדְנֶצַּר מֶלֶךְ־בָּבֶל כסדיא (כַּסְדָּאָה) וּבַיְתָה דְנָה סַתְרֵהּ וְעַמָּה הַגְלִי לְבָבֶל׃  {ס}

ה,יג בְּרַם בִּשְׁנַת חֲדָה לְכוֹרֶשׁ מַלְכָּא דִּי בָבֶל כּוֹרֶשׁ מַלְכָּא שָׂם טְעֵם בֵּית־אֱלָהָא דְנָה לִבְּנֵא׃  ה,יד וְאַף מָאנַיָּא דִי־בֵית־אֱלָהָא דִּי דַהֲבָה וְכַסְפָּא דִּי נְבוּכַדְנֶצַּר הַנְפֵּק מִן־הֵיכְלָא דִּי בִירוּשְׁלֶם וְהֵיבֵל הִמּוֹ לְהֵיכְלָא דִּי בָבֶל הַנְפֵּק הִמּוֹ כּוֹרֶשׁ מַלְכָּא מִן־הֵיכְלָא דִּי בָבֶל וִיהִיבוּ לְשֵׁשְׁבַּצַּר שְׁמֵהּ דִּי פֶחָה שָׂמֵהּ׃  ה,טו וַאֲמַר־לֵהּ אלה (אֵל) מָאנַיָּא שֵׂא אֵזֶל־אֲחֵת הִמּוֹ בְּהֵיכְלָא דִּי בִירוּשְׁלֶם וּבֵית אֱלָהָא יִתְבְּנֵא עַל־אַתְרֵהּ׃  {ס}

ה,טז אֱדַיִן שֵׁשְׁבַּצַּר דֵּךְ אֲתָא יְהַב אֻשַּׁיָּא דִּי־בֵית אֱלָהָא דִּי בִירוּשְׁלֶם וּמִן־אֱדַיִן וְעַד־כְּעַן מִתְבְּנֵא וְלָא שְׁלִם׃  ה,יז וּכְעַן הֵן עַל־מַלְכָּא טָב יִתְבַּקַּר בְּבֵית גִּנְזַיָּא דִּי־מַלְכָּא תַמָּה דִּי בְּבָבֶל הֵן אִיתַי דִּי־מִן־כּוֹרֶשׁ מַלְכָּא שִׂים טְעֵם לְמִבְנֵא בֵּית־אֱלָהָא דֵךְ בִּירוּשְׁלֶם וּרְעוּת מַלְכָּא עַל־דְּנָה יִשְׁלַח עֲלֶינָא׃  {ס}

ו,א בֵּאדַיִן דָּרְיָוֶשׁ מַלְכָּא שָׂם טְעֵם וּבַקַּרוּ בְּבֵית סִפְרַיָּא דִּי גִנְזַיָּא מְהַחֲתִין תַּמָּה בְּבָבֶל׃  ו,ב וְהִשְׁתְּכַח בְּאַחְמְתָא בְּבִירְתָא דִּי בְּמָדַי מְדִינְתָּא מְגִלָּה חֲדָה וְכֵן־כְּתִיב בְּגַוַּהּ דִּכְרוֹנָה׃  {פ}

ו,ג בִּשְׁנַת חֲדָה לְכוֹרֶשׁ מַלְכָּא כּוֹרֶשׁ מַלְכָּא שָׂם טְעֵם בֵּית־אֱלָהָא בִירוּשְׁלֶם בַּיְתָא יִתְבְּנֵא אֲתַר דִּי־דָבְחִין דִּבְחִין וְאֻשּׁוֹהִי מְסוֹבְלִין רוּמֵהּ אַמִּין שִׁתִּין פְּתָיֵהּ אַמִּין שִׁתִּין׃  ו,ד נִדְבָּכִין דִּי־אֶבֶן גְּלָל תְּלָתָא וְנִדְבָּךְ דִּי־אָע חֲדַת וְנִפְקְתָא מִן־בֵּית מַלְכָּא תִּתְיְהִב׃  ו,ה וְאַף מָאנֵי בֵית־אֱלָהָא דִּי דַהֲבָה וְכַסְפָּא דִּי נְבוּכַדְנֶצַּר הַנְפֵּק מִן־הֵיכְלָא דִי־בִירוּשְׁלֶם וְהֵיבֵל לְבָבֶל יַהֲתִיבוּן וִיהָךְ לְהֵיכְלָא דִי־בִירוּשְׁלֶם לְאַתְרֵהּ וְתַחֵת בְּבֵית אֱלָהָא׃  {ס}

ו,ו כְּעַן תַּתְּנַי פַּחַת עֲבַר־נַהֲרָה שְׁתַר בּוֹזְנַי וּכְנָוָתְהוֹן אֲפַרְסְכָיֵא דִּי בַּעֲבַר נַהֲרָה רַחִיקִין הֲווֹ מִן־תַּמָּה׃  ו,ז שְׁבֻקוּ לַעֲבִידַת בֵּית־אֱלָהָא דֵךְ פַּחַת יְהוּדָיֵא וּלְשָׂבֵי יְהוּדָיֵא בֵּית־אֱלָהָא דֵךְ יִבְנוֹן עַל־אַתְרֵהּ׃  ו,ח וּמִנִּי שִׂים טְעֵם לְמָא דִי־תַעַבְדוּן עִם־שָׂבֵי יְהוּדָיֵא אִלֵּךְ לְמִבְנֵא בֵּית־אֱלָהָא דֵךְ וּמִנִּכְסֵי מַלְכָּא דִּי מִדַּת עֲבַר נַהֲרָה אָסְפַּרְנָא נִפְקְתָא תֶּהֱוֵא מִתְיַהֲבָא לְגֻבְרַיָּא אִלֵּךְ דִּי־לָא לְבַטָּלָא׃  ו,ט וּמָה חַשְׁחָן וּבְנֵי תוֹרִין וְדִכְרִין וְאִמְּרִין לַעֲלָוָן לֶאֱלָהּ שְׁמַיָּא חִנְטִין מְלַח חֲמַר וּמְשַׁח כְּמֵאמַר כָּהֲנַיָּא דִי־בִירוּשְׁלֶם לֶהֱוֵא מִתְיְהֵב לְהֹם יוֹם בְּיוֹם דִּי־לָא שָׁלוּ׃  ו,י דִּי־לֶהֱו‍ֹן מְהַקְרְבִין נִיחוֹחִין לֶאֱלָהּ שְׁמַיָּא וּמְצַלַּיִן לְחַיֵּי מַלְכָּא וּבְנוֹהִי׃  ו,יא וּמִנִּי שִׂים טְעֵם דִּי כָל־אֱנָשׁ דִּי יְהַשְׁנֵא פִּתְגָמָא דְנָה יִתְנְסַח אָע מִן־בַּיְתֵהּ וּזְקִיף יִתְמְחֵא עֲלֹהִי וּבַיְתֵהּ נְוָלוּ יִתְעֲבֵד עַל־דְּנָה׃  ו,יב וֵאלָהָא דִּי שַׁכִּן שְׁמֵהּ תַּמָּה יְמַגַּר כָּל־מֶלֶךְ וְעַם דִּי יִשְׁלַח יְדֵהּ לְהַשְׁנָיָה לְחַבָּלָה בֵּית־אֱלָהָא דֵךְ דִּי בִירוּשְׁלֶם אֲנָה דָרְיָוֶשׁ שָׂמֶת טְעֵם אָסְפַּרְנָא יִתְעֲבִד׃  {פ}

ו,יג אֱדַיִן תַּתְּנַי פַּחַת עֲבַר־נַהֲרָה שְׁתַר בּוֹזְנַי וּכְנָוָתְהוֹן  לָקֳבֵל דִּי־שְׁלַח דָּרְיָוֶשׁ מַלְכָּא כְּנֵמָא אָסְפַּרְנָא עֲבַדוּ׃  ו,יד וְשָׂבֵי יְהוּדָיֵא בָּנַיִן וּמַצְלְחִין בִּנְבוּאַת חַגַּי נְבִיָּאה וּזְכַרְיָה בַּר־עִדּוֹא וּבְנוֹ וְשַׁכְלִלוּ מִן־טַעַם אֱלָהּ יִשְׂרָאֵל וּמִטְּעֵם כּוֹרֶשׁ וְדָרְיָוֶשׁ וְאַרְתַּחְשַׁשְׂתְּא מֶלֶךְ פָּרָס׃  ו,טו וְשֵׁיצִיא בַּיְתָה דְנָה עַד יוֹם תְּלָתָה לִירַח אֲדָר דִּי־הִיא שְׁנַת־שֵׁת לְמַלְכוּת דָּרְיָוֶשׁ מַלְכָּא׃  {פ}

ו,טז וַעֲבַדוּ בְנֵי־יִשְׂרָאֵל כָּהֲנַיָּא וְלֵוָיֵא וּשְׁאָר בְּנֵי־גָלוּתָא חֲנֻכַּת בֵּית־אֱלָהָא דְנָה בְּחֶדְוָה׃  ו,יז וְהַקְרִבוּ לַחֲנֻכַּת בֵּית־אֱלָהָא דְנָה תּוֹרִין מְאָה דִּכְרִין מָאתַיִן אִמְּרִין אַרְבַּע מְאָה וּצְפִירֵי עִזִּין לחטיא (לְחַטָּאָה) עַל־כָּל־יִשְׂרָאֵל תְּרֵי־עֲשַׂר לְמִנְיָן שִׁבְטֵי יִשְׂרָאֵל׃  ו,יח וַהֲקִימוּ כָהֲנַיָּא בִּפְלֻגָּתְהוֹן וְלֵוָיֵא בְּמַחְלְקָתְהוֹן עַל־עֲבִידַת אֱלָהָא דִּי בִירוּשְׁלֶם  כִּכְתָב סְפַר מֹשֶׁה׃  {פ}

ו,יט וַיַּעֲשׂוּ בְנֵי־הַגּוֹלָה אֶת־הַפָּסַח בְּאַרְבָּעָה עָשָׂר לַחֹדֶשׁ הָרִאשׁוֹן׃  ו,כ כִּי הִטַּהֲרוּ הַכֹּהֲנִים וְהַלְוִיִּם כְּאֶחָד כֻּלָּם טְהוֹרִים וַיִּשְׁחֲטוּ הַפֶּסַח לְכָל־בְּנֵי הַגּוֹלָה וְלַאֲחֵיהֶם הַכֹּהֲנִים וְלָהֶם׃  ו,כא וַיֹּאכְלוּ בְנֵי־יִשְׂרָאֵל הַשָּׁבִים מֵהַגּוֹלָה וְכֹל הַנִּבְדָּל מִטֻּמְאַת גּוֹיֵ־הָאָרֶץ אֲלֵהֶם לִדְרֹשׁ לַיהוָה אֱלֹהֵי יִשְׂרָאֵל׃  ו,כב וַיַּעֲשׂוּ חַג־מַצּוֹת שִׁבְעַת יָמִים בְּשִׂמְחָה  כִּי שִׂמְּחָם יְהוָה וְהֵסֵב לֵב מֶלֶךְ־אַשּׁוּר עֲלֵיהֶם לְחַזֵּק יְדֵיהֶם בִּמְלֶאכֶת בֵּית־הָאֱלֹהִים אֱלֹהֵי יִשְׂרָאֵל׃  {פ}

ז,א וְאַחַר הַדְּבָרִים הָאֵלֶּה בְּמַלְכוּת אַרְתַּחְשַׁסְתְּא מֶלֶךְ־פָּרָס עֶזְרָא בֶּן־שְׂרָיָה בֶּן־עֲזַרְיָה בֶּן־חִלְקִיָּה׃  ז,ב בֶּן־שַׁלּוּם בֶּן־צָדוֹק בֶּן־אֲחִיטוּב׃  ז,ג בֶּן־אֲמַרְיָה בֶן־עֲזַרְיָה בֶּן־מְרָיוֹת׃  ז,ד בֶּן־זְרַחְיָה בֶן־עֻזִּי בֶּן־בֻּקִּי׃  ז,ה בֶּן־אֲבִישׁוּעַ בֶּן־פִּינְחָס בֶּן־אֶלְעָזָר בֶּן־אַהֲרֹן הַכֹּהֵן הָרֹאשׁ׃  ז,ו הוּא עֶזְרָא עָלָה מִבָּבֶל וְהוּא־סֹפֵר מָהִיר בְּתוֹרַת מֹשֶׁה אֲשֶׁר־נָתַן יְהוָה אֱלֹהֵי יִשְׂרָאֵל וַיִּתֶּן־לוֹ הַמֶּלֶךְ כְּיַד־יְהוָה אֱלֹהָיו עָלָיו כֹּל בַּקָּשָׁתוֹ׃  {פ}

ז,ז וַיַּעֲלוּ מִבְּנֵי־יִשְׂרָאֵל וּמִן־הַכֹּהֲנִים וְהַלְוִיִּם וְהַמְשֹׁרְרִים וְהַשֹּׁעֲרִים וְהַנְּתִינִים אֶל־יְרוּשָׁלִָם  בִּשְׁנַת־שֶׁבַע לְאַרְתַּחְשַׁסְתְּא הַמֶּלֶךְ׃  ז,ח וַיָּבֹא יְרוּשָׁלִַם בַּחֹדֶשׁ הַחֲמִישִׁי  הִיא שְׁנַת הַשְּׁבִיעִית לַמֶּלֶךְ׃  ז,ט כִּי בְּאֶחָד לַחֹדֶשׁ הָרִאשׁוֹן הוּא יְסֻד הַמַּעֲלָה מִבָּבֶל וּבְאֶחָד לַחֹדֶשׁ הַחֲמִישִׁי בָּא אֶל־יְרוּשָׁלִַם כְּיַד־אֱלֹהָיו הַטּוֹבָה עָלָיו׃  ז,י כִּי עֶזְרָא הֵכִין לְבָבוֹ לִדְרֹשׁ אֶת־תּוֹרַת יְהוָה וְלַעֲשֹׂת וּלְלַמֵּד בְּיִשְׂרָאֵל חֹק וּמִשְׁפָּט׃  {ס}

ז,יא וְזֶה פַּרְשֶׁגֶן הַנִּשְׁתְּוָן אֲשֶׁר נָתַן הַמֶּלֶךְ אַרְתַּחְשַׁסְתְּא לְעֶזְרָא הַכֹּהֵן הַסֹּפֵר  סֹפֵר דִּבְרֵי מִצְו‍ֹת־יְהוָה וְחֻקָּיו עַל־יִשְׂרָאֵל׃  {פ}

ז,יב אַרְתַּחְשַׁסְתְּא מֶלֶךְ מַלְכַיָּא  לְעֶזְרָא כָהֲנָא סָפַר דָּתָא דִּי־אֱלָהּ שְׁמַיָּא גְּמִיר וּכְעֶנֶת׃  ז,יג מִנִּי שִׂים טְעֵם דִּי כָל־מִתְנַדַּב בְּמַלְכוּתִי מִן־עַמָּא יִשְׂרָאֵל וְכָהֲנוֹהִי וְלֵוָיֵא לִמְהָךְ לִירוּשְׁלֶם עִמָּךְ יְהָךְ׃  ז,יד כָּל־קֳבֵל דִּי מִן־קֳדָם מַלְכָּא וְשִׁבְעַת יָעֲטֹהִי שְׁלִיחַ לְבַקָּרָה עַל־יְהוּד וְלִירוּשְׁלֶם בְּדָת אֱלָהָךְ דִּי בִידָךְ׃  ז,טו וּלְהֵיבָלָה כְּסַף וּדְהַב דִּי־מַלְכָּא וְיָעֲטוֹהִי הִתְנַדַּבוּ לֶאֱלָהּ יִשְׂרָאֵל דִּי בִירוּשְׁלֶם מִשְׁכְּנֵהּ׃  ז,טז וְכֹל כְּסַף וּדְהַב דִּי תְהַשְׁכַּח בְּכֹל מְדִינַת בָּבֶל עִם הִתְנַדָּבוּת עַמָּא וְכָהֲנַיָּא מִתְנַדְּבִין לְבֵית אֱלָהֲהֹם דִּי בִירוּשְׁלֶם׃  ז,יז כָּל־קֳבֵל דְּנָה אָסְפַּרְנָא תִקְנֵא בְּכַסְפָּא דְנָה תּוֹרִין דִּכְרִין אִמְּרִין וּמִנְחָתְהוֹן וְנִסְכֵּיהוֹן וּתְקָרֵב הִמּוֹ עַל־מַדְבְּחָה דִּי בֵּית אֱלָהֲכֹם דִּי בִירוּשְׁלֶם׃  ז,יח וּמָה דִי עליך (עֲלָךְ) וְעַל־אחיך (אֶחָךְ) יֵיטַב בִּשְׁאָר כַּסְפָּא וְדַהֲבָה לְמֶעְבַּד  כִּרְעוּת אֱלָהֲכֹם תַּעַבְדוּן׃  ז,יט וּמָאנַיָּא דִּי־מִתְיַהֲבִין לָךְ לְפָלְחָן בֵּית אֱלָהָךְ הַשְׁלֵם קֳדָם אֱלָהּ יְרוּשְׁלֶם׃  ז,כ וּשְׁאָר חַשְׁחוּת בֵּית אֱלָהָךְ דִּי יִפֶּל־לָךְ לְמִנְתַּן תִּנְתֵּן מִן־בֵּית גִּנְזֵי מַלְכָּא׃  ז,כא וּמִנִּי אֲנָה אַרְתַּחְשַׁסְתְּא מַלְכָּא שִׂים טְעֵם לְכֹל גִּזַּבְרַיָּא דִּי בַּעֲבַר נַהֲרָה  דִּי כָל־דִּי יִשְׁאֲלֶנְכוֹן עֶזְרָא כָהֲנָא סָפַר דָּתָא דִּי־אֱלָהּ שְׁמַיָּא אָסְפַּרְנָא יִתְעֲבִד׃  ז,כב עַד־כְּסַף כַּכְּרִין מְאָה וְעַד־חִנְטִין כּוֹרִין מְאָה וְעַד־חֲמַר בַּתִּין מְאָה וְעַד־בַּתִּין מְשַׁח מְאָה וּמְלַח דִּי־לָא כְתָב׃  ז,כג כָּל־דִּי מִן־טַעַם אֱלָהּ שְׁמַיָּא יִתְעֲבֵד אַדְרַזְדָּא לְבֵית אֱלָהּ שְׁמַיָּא  דִּי־לְמָה לֶהֱוֵא קְצַף עַל־מַלְכוּת מַלְכָּא וּבְנוֹהִי׃  ז,כד וּלְכֹם מְהוֹדְעִין דִּי כָל־כָּהֲנַיָּא וְלֵוָיֵא זַמָּרַיָּא תָרָעַיָּא נְתִינַיָּא וּפָלְחֵי בֵּית אֱלָהָא דְנָה מִנְדָּה בְלוֹ וַהֲלָךְ לָא שַׁלִּיט לְמִרְמֵא עֲלֵיהֹם׃  ז,כה וְאַנְתְּ עֶזְרָא כְּחָכְמַת אֱלָהָךְ דִּי־בִידָךְ מֶנִּי שָׁפְטִין וְדַיָּנִין דִּי־לֶהֱו‍ֹן דאנין (דָּיְנִין) לְכָל־עַמָּא דִּי בַּעֲבַר נַהֲרָה לְכָל־יָדְעֵי דָּתֵי אֱלָהָךְ וְדִי לָא יָדַע תְּהוֹדְעוּן׃  ז,כו וְכָל־דִּי־לָא לֶהֱוֵא עָבֵד דָּתָא דִי־אֱלָהָךְ וְדָתָא דִּי מַלְכָּא אָסְפַּרְנָא דִּינָה לֶהֱוֵא מִתְעֲבֵד מִנֵּהּ  הֵן לְמוֹת הֵן לשרשו (לִשְׁרֹשִׁי) הֵן־לַעֲנָשׁ נִכְסִין וְלֶאֱסוּרִין׃  {פ}

ז,כז בָּרוּךְ יְהוָה אֱלֹהֵי אֲבֹתֵינוּ אֲשֶׁר נָתַן כָּזֹאת בְּלֵב הַמֶּלֶךְ לְפָאֵר אֶת־בֵּית יְהוָה אֲשֶׁר בִּירוּשָׁלִָם׃  ז,כח וְעָלַי הִטָּה־חֶסֶד לִפְנֵי הַמֶּלֶךְ וְיוֹעֲצָיו וּלְכָל־שָׂרֵי הַמֶּלֶךְ הַגִּבֹּרִים וַאֲנִי הִתְחַזַּקְתִּי כְּיַד־יְהוָה אֱלֹהַי עָלַי וָאֶקְבְּצָה מִיִּשְׂרָאֵל רָאשִׁים לַעֲלוֹת עִמִּי׃  {פ}

ח,א וְאֵלֶּה רָאשֵׁי אֲבֹתֵיהֶם וְהִתְיַחְשָׂם  הָעֹלִים עִמִּי בְּמַלְכוּת אַרְתַּחְשַׁסְתְּא הַמֶּלֶךְ מִבָּבֶל׃  {ס}  ח,ב מִבְּנֵי פִינְחָס גֵּרְשֹׁם  {ס}  מִבְּנֵי אִיתָמָר דָּנִיֵּאל  {ס}  מִבְּנֵי דָוִיד חַטּוּשׁ׃  {ס}  ח,ג מִבְּנֵי שְׁכַנְיָה  {ס}  מִבְּנֵי פַרְעֹשׁ זְכַרְיָה וְעִמּוֹ הִתְיַחֵשׂ לִזְכָרִים מֵאָה וַחֲמִשִּׁים׃  {ס}  ח,ד מִבְּנֵי פַּחַת מוֹאָב אֶלְיְהוֹעֵינַי בֶּן־זְרַחְיָה וְעִמּוֹ מָאתַיִם הַזְּכָרִים׃  {ס}  ח,ה מִבְּנֵי שְׁכַנְיָה בֶּן־יַחֲזִיאֵל וְעִמּוֹ שְׁלֹשׁ מֵאוֹת הַזְּכָרִים׃  {ס}  ח,ו וּמִבְּנֵי עָדִין עֶבֶד בֶּן־יוֹנָתָן וְעִמּוֹ חֲמִשִּׁים הַזְּכָרִים׃  {ס}  ח,ז וּמִבְּנֵי עֵילָם יְשַׁעְיָה בֶּן־עֲתַלְיָה וְעִמּוֹ שִׁבְעִים הַזְּכָרִים׃  {ס}  ח,ח וּמִבְּנֵי שְׁפַטְיָה זְבַדְיָה בֶּן־מִיכָאֵל וְעִמּוֹ שְׁמֹנִים הַזְּכָרִים׃  {ס}  ח,ט מִבְּנֵי יוֹאָב עֹבַדְיָה בֶּן־יְחִיאֵל וְעִמּוֹ מָאתַיִם וּשְׁמֹנָה עָשָׂר הַזְּכָרִים׃  {ס}  ח,י וּמִבְּנֵי שְׁלוֹמִית בֶּן־יוֹסִפְיָה וְעִמּוֹ מֵאָה וְשִׁשִּׁים הַזְּכָרִים׃  {ס}  ח,יא וּמִבְּנֵי בֵבַי זְכַרְיָה בֶּן־בֵּבָי וְעִמּוֹ עֶשְׂרִים וּשְׁמֹנָה הַזְּכָרִים׃  {ס}  ח,יב וּמִבְּנֵי עַזְגָּד יוֹחָנָן בֶּן־הַקָּטָן וְעִמּוֹ מֵאָה וַעֲשָׂרָה הַזְּכָרִים׃  {ס}  ח,יג וּמִבְּנֵי אֲדֹנִיקָם אַחֲרֹנִים וְאֵלֶּה שְׁמוֹתָם אֱלִיפֶלֶט יְעִיאֵל וּשְׁמַעְיָה וְעִמָּהֶם שִׁשִּׁים הַזְּכָרִים׃  {ס}  ח,יד וּמִבְּנֵי בִגְוַי עוּתַי וזבוד (וְזַכּוּר) וְעִמּוֹ שִׁבְעִים הַזְּכָרִים׃  {פ}

ח,טו וָאֶקְבְּצֵם אֶל־הַנָּהָר הַבָּא אֶל־אַהֲוָא וַנַּחֲנֶה שָׁם יָמִים שְׁלֹשָׁה וָאָבִינָה בָעָם וּבַכֹּהֲנִים וּמִבְּנֵי לֵוִי לֹא־מָצָאתִי שָׁם׃  ח,טז וָאֶשְׁלְחָה לֶאֱלִיעֶזֶר לַאֲרִיאֵל לִשְׁמַעְיָה וּלְאֶלְנָתָן וּלְיָרִיב וּלְאֶלְנָתָן וּלְנָתָן וְלִזְכַרְיָה וְלִמְשֻׁלָּם רָאשִׁים וּלְיוֹיָרִיב וּלְאֶלְנָתָן מְבִינִים׃  ח,יז ואוצאה (וָאֲצַוֶּה) אוֹתָם עַל־אִדּוֹ הָרֹאשׁ בְּכָסִפְיָא הַמָּקוֹם וָאָשִׂימָה בְּפִיהֶם דְּבָרִים לְדַבֵּר אֶל־אִדּוֹ אָחִיו הנתונים (הַנְּתִינִים) בְּכָסִפְיָא הַמָּקוֹם לְהָבִיא־לָנוּ מְשָׁרְתִים לְבֵית אֱלֹהֵינוּ׃  ח,יח וַיָּבִיאּוּ לָנוּ כְּיַד־אֱלֹהֵינוּ הַטּוֹבָה עָלֵינוּ אִישׁ שֶׂכֶל מִבְּנֵי מַחְלִי בֶּן־לֵוִי בֶּן־יִשְׂרָאֵל וְשֵׁרֵבְיָה וּבָנָיו וְאֶחָיו שְׁמֹנָה עָשָׂר׃  ח,יט וְאֶת־חֲשַׁבְיָה וְאִתּוֹ יְשַׁעְיָה מִבְּנֵי מְרָרִי  אֶחָיו וּבְנֵיהֶם עֶשְׂרִים׃  {ס}

ח,כ וּמִן־הַנְּתִינִים שֶׁנָּתַן דָּוִיד וְהַשָּׂרִים לַעֲבֹדַת הַלְוִיִּם נְתִינִים מָאתַיִם וְעֶשְׂרִים  כֻּלָּם נִקְּבוּ בְשֵׁמוֹת׃  ח,כא וָאֶקְרָא שָׁם צוֹם עַל־הַנָּהָר אַהֲוָא לְהִתְעַנּוֹת לִפְנֵי אֱלֹהֵינוּ לְבַקֵּשׁ מִמֶּנּוּ דֶּרֶךְ יְשָׁרָה לָנוּ וּלְטַפֵּנוּ וּלְכָל־רְכוּשֵׁנוּ׃  ח,כב כִּי בֹשְׁתִּי לִשְׁאוֹל מִן־הַמֶּלֶךְ חַיִל וּפָרָשִׁים לְעָזְרֵנוּ מֵאוֹיֵב בַּדָּרֶךְ  כִּי־אָמַרְנוּ לַמֶּלֶךְ לֵאמֹר יַד־אֱלֹהֵינוּ עַל־כָּל־מְבַקְשָׁיו לְטוֹבָה וְעֻזּוֹ וְאַפּוֹ עַל כָּל־עֹזְבָיו׃  ח,כג וַנָּצוּמָה וַנְּבַקְשָׁה מֵאֱלֹהֵינוּ עַל־זֹאת וַיֵּעָתֵר לָנוּ׃  ח,כד וָאַבְדִּילָה מִשָּׂרֵי הַכֹּהֲנִים שְׁנֵים עָשָׂר לְשֵׁרֵבְיָה חֲשַׁבְיָה וְעִמָּהֶם מֵאֲחֵיהֶם עֲשָׂרָה׃  ח,כה ואשקולה (וָאֶשְׁקֳלָה) לָהֶם אֶת־הַכֶּסֶף וְאֶת־הַזָּהָב וְאֶת־הַכֵּלִים תְּרוּמַת בֵּית־אֱלֹהֵינוּ הַהֵרִימוּ הַמֶּלֶךְ וְיֹעֲצָיו וְשָׂרָיו וְכָל־יִשְׂרָאֵל הַנִּמְצָאִים׃  ח,כו וָאֶשְׁקְלָה עַל־יָדָם כֶּסֶף כִּכָּרִים שֵׁשׁ־מֵאוֹת וַחֲמִשִּׁים וּכְלֵי־כֶסֶף מֵאָה לְכִכָּרִים זָהָב מֵאָה כִכָּר׃  ח,כז וּכְפֹרֵי זָהָב עֶשְׂרִים לַאֲדַרְכֹנִים אָלֶף וּכְלֵי נְחֹשֶׁת מֻצְהָב טוֹבָה שְׁנַיִם חֲמוּדֹת כַּזָּהָב׃  ח,כח וָאֹמְרָה אֲלֵהֶם אַתֶּם קֹדֶשׁ לַיהוָה וְהַכֵּלִים קֹדֶשׁ וְהַכֶּסֶף וְהַזָּהָב נְדָבָה לַיהוָה אֱלֹהֵי אֲבֹתֵיכֶם׃  ח,כט שִׁקְדוּ וְשִׁמְרוּ עַד־תִּשְׁקְלוּ לִפְנֵי שָׂרֵי הַכֹּהֲנִים וְהַלְוִיִּם וְשָׂרֵי־הָאָבוֹת לְיִשְׂרָאֵל בִּירוּשָׁלִָם  הַלִּשְׁכוֹת בֵּית יְהוָה׃  ח,ל וְקִבְּלוּ הַכֹּהֲנִים וְהַלְוִיִּם מִשְׁקַל הַכֶּסֶף וְהַזָּהָב וְהַכֵּלִים לְהָבִיא לִירוּשָׁלִַם לְבֵית אֱלֹהֵינוּ׃  {פ}

ח,לא וַנִּסְּעָה מִנְּהַר אַהֲוָא בִּשְׁנֵים עָשָׂר לַחֹדֶשׁ הָרִאשׁוֹן לָלֶכֶת יְרוּשָׁלִָם וְיַד־אֱלֹהֵינוּ הָיְתָה עָלֵינוּ וַיַּצִּילֵנוּ מִכַּף אוֹיֵב וְאוֹרֵב עַל־הַדָּרֶךְ׃  ח,לב וַנָּבוֹא יְרוּשָׁלִָם וַנֵּשֶׁב שָׁם יָמִים שְׁלֹשָׁה׃  ח,לג וּבַיּוֹם הָרְבִיעִי נִשְׁקַל הַכֶּסֶף וְהַזָּהָב וְהַכֵּלִים בְּבֵית אֱלֹהֵינוּ עַל יַד־מְרֵמוֹת בֶּן־אוּרִיָּה הַכֹּהֵן וְעִמּוֹ אֶלְעָזָר בֶּן־פִּינְחָס וְעִמָּהֶם יוֹזָבָד בֶּן־יֵשׁוּעַ וְנוֹעַדְיָה בֶן־בִּנּוּי הַלְוִיִּם׃  ח,לד בְּמִסְפָּר בְּמִשְׁקָל לַכֹּל וַיִּכָּתֵב כָּל־הַמִּשְׁקָל בָּעֵת הַהִיא׃  {פ}

ח,לה הַבָּאִים מֵהַשְּׁבִי בְנֵי־הַגּוֹלָה הִקְרִיבוּ עֹלוֹת לֵאלֹהֵי יִשְׂרָאֵל פָּרִים שְׁנֵים־עָשָׂר עַל־כָּל־יִשְׂרָאֵל אֵילִים תִּשְׁעִים וְשִׁשָּׁה כְּבָשִׂים שִׁבְעִים וְשִׁבְעָה צְפִירֵי חַטָּאת שְׁנֵים עָשָׂר  הַכֹּל עוֹלָה לַיהוָה׃  {פ}

ח,לו וַיִּתְּנוּ אֶת־דָּתֵי הַמֶּלֶךְ לַאֲחַשְׁדַּרְפְּנֵי הַמֶּלֶךְ וּפַחֲווֹת עֵבֶר הַנָּהָר וְנִשְּׂאוּ אֶת־הָעָם וְאֶת־בֵּית הָאֱלֹהִים׃  {ס}

ט,א וּכְכַלּוֹת אֵלֶּה נִגְּשׁוּ אֵלַי הַשָּׂרִים לֵאמֹר לֹא־נִבְדְּלוּ הָעָם יִשְׂרָאֵל וְהַכֹּהֲנִים וְהַלְוִיִּם מֵעַמֵּי הָאֲרָצוֹת  כְּתֹעֲבֹתֵיהֶם לַכְּנַעֲנִי הַחִתִּי הַפְּרִזִּי הַיְבוּסִי הָעַמֹּנִי הַמֹּאָבִי הַמִּצְרִי וְהָאֱמֹרִי׃  ט,ב כִּי־נָשְׂאוּ מִבְּנֹתֵיהֶם לָהֶם וְלִבְנֵיהֶם וְהִתְעָרְבוּ זֶרַע הַקֹּדֶשׁ בְּעַמֵּי הָאֲרָצוֹת וְיַד הַשָּׂרִים וְהַסְּגָנִים הָיְתָה בַּמַּעַל הַזֶּה רִאשׁוֹנָה׃  {ס}

ט,ג וּכְשָׁמְעִי אֶת־הַדָּבָר הַזֶּה קָרַעְתִּי אֶת־בִּגְדִי וּמְעִילִי וָאֶמְרְטָה מִשְּׂעַר רֹאשִׁי וּזְקָנִי וָאֵשְׁבָה מְשׁוֹמֵם׃  ט,ד וְאֵלַי יֵאָסְפוּ כֹּל חָרֵד בְּדִבְרֵי אֱלֹהֵי־יִשְׂרָאֵל עַל מַעַל הַגּוֹלָה וַאֲנִי יֹשֵׁב מְשׁוֹמֵם עַד לְמִנְחַת הָעָרֶב׃  ט,ה וּבְמִנְחַת הָעֶרֶב קַמְתִּי מִתַּעֲנִיתִי וּבְקָרְעִי בִגְדִי וּמְעִילִי וָאֶכְרְעָה עַל־בִּרְכַּי וָאֶפְרְשָׂה כַפַּי אֶל־יְהוָה אֱלֹהָי׃  ט,ו וָאֹמְרָה אֱלֹהַי בֹּשְׁתִּי וְנִכְלַמְתִּי לְהָרִים אֱלֹהַי פָּנַי אֵלֶיךָ  כִּי עֲו‍ֹנֹתֵינוּ רָבוּ לְמַעְלָה רֹּאשׁ וְאַשְׁמָתֵנוּ גָדְלָה עַד לַשָּׁמָיִם׃  ט,ז מִימֵי אֲבֹתֵינוּ אֲנַחְנוּ בְּאַשְׁמָה גְדֹלָה עַד הַיּוֹם הַזֶּה וּבַעֲו‍ֹנֹתֵינוּ נִתַּנּוּ אֲנַחְנוּ מְלָכֵינוּ כֹהֲנֵינוּ בְּיַד מַלְכֵי הָאֲרָצוֹת בַּחֶרֶב בַּשְּׁבִי וּבַבִּזָּה וּבְבֹשֶׁת פָּנִים כְּהַיּוֹם הַזֶּה׃  ט,ח וְעַתָּה כִּמְעַט־רֶגַע הָיְתָה תְחִנָּה מֵאֵת יְהוָה אֱלֹהֵינוּ לְהַשְׁאִיר לָנוּ פְּלֵיטָה וְלָתֶת־לָנוּ יָתֵד בִּמְקוֹם קָדְשׁוֹ לְהָאִיר עֵינֵינוּ אֱלֹהֵינוּ וּלְתִתֵּנוּ מִחְיָה מְעַט בְּעַבְדֻתֵנוּ׃  ט,ט כִּי־עֲבָדִים אֲנַחְנוּ וּבְעַבְדֻתֵנוּ לֹא עֲזָבָנוּ אֱלֹהֵינוּ וַיַּט־עָלֵינוּ חֶסֶד לִפְנֵי מַלְכֵי פָרַס לָתֶת־לָנוּ מִחְיָה לְרוֹמֵם אֶת־בֵּית אֱלֹהֵינוּ וּלְהַעֲמִיד אֶת־חָרְבֹתָיו וְלָתֶת־לָנוּ גָדֵר בִּיהוּדָה וּבִירוּשָׁלִָם׃  {ס}

ט,י וְעַתָּה מַה־נֹּאמַר אֱלֹהֵינוּ אַחֲרֵי־זֹאת  כִּי עָזַבְנוּ מִצְו‍ֹתֶיךָ׃  ט,יא אֲשֶׁר צִוִּיתָ בְּיַד עֲבָדֶיךָ הַנְּבִיאִים לֵאמֹר הָאָרֶץ אֲשֶׁר אַתֶּם בָּאִים לְרִשְׁתָּהּ אֶרֶץ נִדָּה הִיא בְּנִדַּת עַמֵּי הָאֲרָצוֹת  בְּתוֹעֲבֹתֵיהֶם אֲשֶׁר מִלְאוּהָ מִפֶּה אֶל־פֶּה בְּטֻמְאָתָם׃  ט,יב וְעַתָּה בְּנוֹתֵיכֶם אַל־תִּתְּנוּ לִבְנֵיהֶם וּבְנֹתֵיהֶם אַל־תִּשְׂאוּ לִבְנֵיכֶם וְלֹא־תִדְרְשׁוּ שְׁלֹמָם וְטוֹבָתָם עַד־עוֹלָם לְמַעַן תֶּחֶזְקוּ וַאֲכַלְתֶּם אֶת־טוּב הָאָרֶץ וְהוֹרַשְׁתֶּם לִבְנֵיכֶם עַד־עוֹלָם׃  ט,יג וְאַחֲרֵי כָּל־הַבָּא עָלֵינוּ בְּמַעֲשֵׂינוּ הָרָעִים וּבְאַשְׁמָתֵנוּ הַגְּדֹלָה  כִּי אַתָּה אֱלֹהֵינוּ חָשַׂכְתָּ לְמַטָּה מֵעֲו‍ֹנֵנוּ וְנָתַתָּה לָּנוּ פְּלֵיטָה כָּזֹאת׃  ט,יד הֲנָשׁוּב לְהָפֵר מִצְו‍ֹתֶיךָ וּלְהִתְחַתֵּן בְּעַמֵּי הַתֹּעֵבוֹת הָאֵלֶּה  הֲלוֹא תֶאֱנַף־בָּנוּ עַד־כַּלֵּה לְאֵין שְׁאֵרִית וּפְלֵיטָה׃  {פ}

ט,טו יְהוָה אֱלֹהֵי יִשְׂרָאֵל צַדִּיק אַתָּה כִּי־נִשְׁאַרְנוּ פְלֵיטָה כְּהַיּוֹם הַזֶּה הִנְנוּ לְפָנֶיךָ בְּאַשְׁמָתֵינוּ כִּי אֵין לַעֲמוֹד לְפָנֶיךָ עַל־זֹאת׃  {פ}

י,א וּכְהִתְפַּלֵּל עֶזְרָא וּכְהִתְוַדֹּתוֹ בֹּכֶה וּמִתְנַפֵּל לִפְנֵי בֵּית הָאֱלֹהִים  נִקְבְּצוּ אֵלָיו מִיִּשְׂרָאֵל קָהָל רַב־מְאֹד אֲנָשִׁים וְנָשִׁים וִילָדִים כִּי־בָכוּ הָעָם הַרְבֵּה־בֶכֶה׃  {ס}

י,ב וַיַּעַן שְׁכַנְיָה בֶן־יְחִיאֵל מִבְּנֵי עולם (עֵילָם) וַיֹּאמֶר לְעֶזְרָא אֲנַחְנוּ מָעַלְנוּ בֵאלֹהֵינוּ וַנֹּשֶׁב נָשִׁים נָכְרִיּוֹת מֵעַמֵּי הָאָרֶץ וְעַתָּה יֵשׁ־מִקְוֶה לְיִשְׂרָאֵל עַל־זֹאת׃  י,ג וְעַתָּה נִכְרָת־בְּרִית לֵאלֹהֵינוּ לְהוֹצִיא כָל־נָשִׁים וְהַנּוֹלָד מֵהֶם בַּעֲצַת אֲדֹנָי וְהַחֲרֵדִים בְּמִצְוַת אֱלֹהֵינוּ וְכַתּוֹרָה יֵעָשֶׂה׃  י,ד קוּם כִּי־עָלֶיךָ הַדָּבָר וַאֲנַחְנוּ עִמָּךְ  חֲזַק וַעֲשֵׂה׃  {פ}

י,ה וַיָּקָם עֶזְרָא וַיַּשְׁבַּע אֶת־שָׂרֵי הַכֹּהֲנִים הַלְוִיִּם וְכָל־יִשְׂרָאֵל לַעֲשׂוֹת כַּדָּבָר הַזֶּה וַיִּשָּׁבֵעוּ׃  י,ו וַיָּקָם עֶזְרָא מִלִּפְנֵי בֵּית הָאֱלֹהִים וַיֵּלֶךְ אֶל־לִשְׁכַּת יְהוֹחָנָן בֶּן־אֶלְיָשִׁיב וַיֵּלֶךְ שָׁם לֶחֶם לֹא־אָכַל וּמַיִם לֹא־שָׁתָה כִּי מִתְאַבֵּל עַל־מַעַל הַגּוֹלָה׃  {ס}

י,ז וַיַּעֲבִירוּ קוֹל בִּיהוּדָה וִירוּשָׁלִַם לְכֹל בְּנֵי הַגּוֹלָה לְהִקָּבֵץ יְרוּשָׁלִָם׃  י,ח וְכֹל אֲשֶׁר לֹא־יָבוֹא לִשְׁלֹשֶׁת הַיָּמִים כַּעֲצַת הַשָּׂרִים וְהַזְּקֵנִים יָחֳרַם כָּל־רְכוּשׁוֹ וְהוּא יִבָּדֵל מִקְּהַל הַגּוֹלָה׃  {ס}

י,ט וַיִּקָּבְצוּ כָל־אַנְשֵׁי־יְהוּדָה וּבִנְיָמִן יְרוּשָׁלִַם לִשְׁלֹשֶׁת הַיָּמִים הוּא חֹדֶשׁ הַתְּשִׁיעִי בְּעֶשְׂרִים בַּחֹדֶשׁ וַיֵּשְׁבוּ כָל־הָעָם בִּרְחוֹב בֵּית הָאֱלֹהִים מַרְעִידִים עַל־הַדָּבָר וּמֵהַגְּשָׁמִים׃  {פ}

י,י וַיָּקָם עֶזְרָא הַכֹּהֵן וַיֹּאמֶר אֲלֵהֶם אַתֶּם מְעַלְתֶּם וַתֹּשִׁיבוּ נָשִׁים נָכְרִיּוֹת לְהוֹסִיף עַל־אַשְׁמַת יִשְׂרָאֵל׃  י,יא וְעַתָּה תְּנוּ תוֹדָה לַיהוָה אֱלֹהֵי־אֲבֹתֵיכֶם וַעֲשׂוּ רְצוֹנוֹ וְהִבָּדְלוּ מֵעַמֵּי הָאָרֶץ וּמִן־הַנָּשִׁים הַנָּכְרִיּוֹת׃  י,יב וַיַּעֲנוּ כָל־הַקָּהָל וַיֹּאמְרוּ קוֹל גָּדוֹל  כֵּן כדבריך (כִּדְבָרְךָ) עָלֵינוּ לַעֲשׂוֹת׃  י,יג אֲבָל הָעָם רָב וְהָעֵת גְּשָׁמִים וְאֵין כֹּחַ לַעֲמוֹד בַּחוּץ וְהַמְּלָאכָה לֹא־לְיוֹם אֶחָד וְלֹא לִשְׁנַיִם כִּי־הִרְבִּינוּ לִפְשֹׁעַ בַּדָּבָר הַזֶּה׃  י,יד יַעַמְדוּ־נָא שָׂרֵינוּ לְכָל־הַקָּהָל וְכֹל אֲשֶׁר בֶּעָרֵינוּ הַהֹשִׁיב נָשִׁים נָכְרִיּוֹת יָבֹא לְעִתִּים מְזֻמָּנִים וְעִמָּהֶם זִקְנֵי־עִיר וָעִיר וְשֹׁפְטֶיהָ  עַד לְהָשִׁיב חֲרוֹן אַף־אֱלֹהֵינוּ מִמֶּנּוּ עַד לַדָּבָר הַזֶּה׃  {פ}

י,טו אַךְ יוֹנָתָן בֶּן־עֲשָׂהאֵל וְיַחְזְיָה בֶן־תִּקְוָה עָמְדוּ עַל־זֹאת וּמְשֻׁלָּם וְשַׁבְּתַי הַלֵּוִי עֲזָרֻם׃  י,טז וַיַּעֲשׂוּ־כֵן בְּנֵי הַגּוֹלָה וַיִּבָּדְלוּ עֶזְרָא הַכֹּהֵן אֲנָשִׁים רָאשֵׁי הָאָבוֹת לְבֵית אֲבֹתָם וְכֻלָּם בְּשֵׁמוֹת וַיֵּשְׁבוּ בְּיוֹם אֶחָד לַחֹדֶשׁ הָעֲשִׂירִי לְדַרְיוֹשׁ הַדָּבָר׃  י,יז וַיְכַלּוּ בַכֹּל אֲנָשִׁים הַהֹשִׁיבוּ נָשִׁים נָכְרִיּוֹת  עַד יוֹם אֶחָד לַחֹדֶשׁ הָרִאשׁוֹן׃  {פ}

י,יח וַיִּמָּצֵא מִבְּנֵי הַכֹּהֲנִים אֲשֶׁר הֹשִׁיבוּ נָשִׁים נָכְרִיּוֹת  מִבְּנֵי יֵשׁוּעַ בֶּן־יוֹצָדָק וְאֶחָיו מַעֲשֵׂיָה וֶאֱלִיעֶזֶר וְיָרִיב וּגְדַלְיָה׃  י,יט וַיִּתְּנוּ יָדָם לְהוֹצִיא נְשֵׁיהֶם וַאֲשֵׁמִים אֵיל־צֹאן עַל־אַשְׁמָתָם׃  {ס}  י,כ וּמִבְּנֵי אִמֵּר חֲנָנִי וּזְבַדְיָה׃  {ס}  י,כא וּמִבְּנֵי חָרִם מַעֲשֵׂיָה וְאֵלִיָּה וּשְׁמַעְיָה וִיחִיאֵל וְעֻזִּיָּה׃  י,כב וּמִבְּנֵי פַּשְׁחוּר אֶלְיוֹעֵינַי מַעֲשֵׂיָה יִשְׁמָעֵאל נְתַנְאֵל יוֹזָבָד וְאֶלְעָשָׂה׃  {ס}  י,כג וּמִן־הַלְוִיִּם יוֹזָבָד וְשִׁמְעִי וְקֵלָיָה הוּא קְלִיטָא פְּתַחְיָה יְהוּדָה וֶאֱלִיעֶזֶר׃  {ס}  י,כד וּמִן־הַמְשֹׁרְרִים אֶלְיָשִׁיב וּמִן־הַשֹּׁעֲרִים שַׁלֻּם וָטֶלֶם וְאוּרִי׃  {ס}  י,כה וּמִיִּשְׂרָאֵל מִבְּנֵי פַרְעֹשׁ רַמְיָה וְיִזִּיָּה וּמַלְכִּיָּה וּמִיָּמִן וְאֶלְעָזָר וּמַלְכִּיָּה וּבְנָיָה׃  {ס}  י,כו וּמִבְּנֵי עֵילָם מַתַּנְיָה זְכַרְיָה וִיחִיאֵל וְעַבְדִּי וִירֵמוֹת וְאֵלִיָּה׃  {ס}  י,כז וּמִבְּנֵי זַתּוּא אֶלְיוֹעֵנַי אֶלְיָשִׁיב מַתַּנְיָה וִירֵמוֹת וְזָבָד וַעֲזִיזָא׃  {ס}  י,כח וּמִבְּנֵי בֵּבָי יְהוֹחָנָן חֲנַנְיָה זַבַּי עַתְלָי׃  {ס}  י,כט וּמִבְּנֵי בָּנִי מְשֻׁלָּם מַלּוּךְ וַעֲדָיָה יָשׁוּב וּשְׁאָל ירמות (וְרָמוֹת)׃  {ס}  י,ל וּמִבְּנֵי פַּחַת מוֹאָב עַדְנָא וּכְלָל בְּנָיָה מַעֲשֵׂיָה מַתַּנְיָה בְצַלְאֵל וּבִנּוּי וּמְנַשֶּׁה׃  {ס}  י,לא וּבְנֵי חָרִם אֱלִיעֶזֶר יִשִּׁיָּה מַלְכִּיָּה שְׁמַעְיָה שִׁמְעוֹן׃  י,לב בִּנְיָמִן מַלּוּךְ שְׁמַרְיָה׃  {ס}  י,לג מִבְּנֵי חָשֻׁם מַתְּנַי מַתַּתָּה זָבָד אֱלִיפֶלֶט יְרֵמַי מְנַשֶּׁה שִׁמְעִי׃  {ס}  י,לד מִבְּנֵי בָנִי מַעֲדַי עַמְרָם וְאוּאֵל׃  {ס}  י,לה בְּנָיָה בֵדְיָה כלוהי (כְּלוּהוּ)׃  י,לו וַנְיָה מְרֵמוֹת אֶלְיָשִׁיב׃  י,לז מַתַּנְיָה מַתְּנַי ויעשו (וְיַעֲשָׂי)׃  י,לח וּבָנִי וּבִנּוּי שִׁמְעִי׃  י,לט וְשֶׁלֶמְיָה וְנָתָן וַעֲדָיָה׃  י,מ מַכְנַדְבַי שָׁשַׁי שָׁרָי׃  י,מא עֲזַרְאֵל וְשֶׁלֶמְיָהוּ שְׁמַרְיָה׃  י,מב שַׁלּוּם אֲמַרְיָה יוֹסֵף׃  {ס}  י,מג מִבְּנֵי נְבוֹ יְעִיאֵל מַתִּתְיָה זָבָד זְבִינָא ידו (יַדַּי) וְיוֹאֵל בְּנָיָה׃  י,מד כָּל־אֵלֶּה נשאי (נָשְׂאוּ) נָשִׁים נָכְרִיּוֹת וְיֵשׁ מֵהֶם נָשִׁים וַיָּשִׂימוּ בָּנִים׃  {פ}
